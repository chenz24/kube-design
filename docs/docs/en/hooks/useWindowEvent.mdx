## Usage

useWindowEvent adds event listener to window object on component mount and removes it on unmount:

```tsx
const handler = (event) => console.log(event);

// regular way
useEffect(() => {
  window.addEventListener('keydown', handler);
  return () => window.removeEventListener('keydown');
}, []);

// with use-window-event hook
useWindowEvent('keydown', handler);
```

## Example

```tsx
function Search() {
  const inputRef = useRef();

  useWindowEvent('keydown', (event) => {
    if (event.code === 'KeyK' && (event.ctrlKey || event.metaKey)) {
      event.preventDefault();
      inputRef.current.focus();
    }
  });

  return <input ref={inputRef} />;
}
```

## Definition

Hook has exact same definition as window.addEventListener function:

```tsx
function useWindowEvent<K extends keyof WindowEventMap>(
  type: K,
  listener: (this: Window, ev: WindowEventMap[K]) => any,
  options?: boolean | AddEventListenerOptions
): void;
```
