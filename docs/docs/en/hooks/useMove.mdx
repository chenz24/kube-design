## Usage

use-move handles move behavior over any element:

```js live=true
() => {
  const [value, setValue] = React.useState({ x: 0.2, y: 0.6 });
  const { ref, active } = useMove(setValue);

  return (
    <>
      <div
        ref={ref}
        style={{
          width: 400,
          height: 120,
          backgroundColor: "#6b7b95",
          position: 'relative',
        }}
      >
        <div
          style={{
            position: 'absolute',
            left: `calc(${value.x * 100}% - 8px)`,
            top: `calc(${value.y * 100}% - 8px)`,
            width: 16,
            height: 16,
            backgroundColor: active ? "#55bc8a" : "#329dce",
          }}
        />
      </div>

      <Text>
        Values <Card>{`{ x: ${Math.round(value.x * 100)}, y: ${Math.round(value.y * 100)} }`}</Card>
      </Text>
    </>
  );
}
```

## API

Hook accepts callback that is called when user moves pressed mouse over given element and returns object with ref and active state:

```tsx
const {
  ref, // -> pass ref to target element
  active, // -> is user changing something right now?
} = useMove(({ x, y }) => setValue({ x, y }));
```

## Examples

## Horizontal slider

```js live=true
() => {
  const [value, setValue] = React.useState(0.2);
  const { ref } = useMove(({ x }) => setValue(x));

  return (
    <>
      <Group position="center">
        <div
          ref={ref}
          style={{
            width: 400,
            height: 16,
            backgroundColor: "#6b7b95",
            position: 'relative',
          }}
        >
          {/* Filled bar */}
          <div
            style={{
              width: `${value * 100}%`,
              height: 16,
              backgroundColor: "#6b7b95",
            }}
          />

          {/* Thumb */}
          <div
            style={{
              position: 'absolute',
              left: `calc(${value * 100}% - 8px)`,
              top: 0,
              width: 16,
              height: 16,
              backgroundColor:  "#329dce",
            }}
          />
        </div>
      </Group>

      <Text align="center" style={{ marginTop: '10px' }}>
        Value: {Math.round(value * 100)}
      </Text>
    </>
  );
}
```

## Color picker

```js live=true
() => {
  const [value, setValue] = React.useState({ x: 0.2, y: 0.6 });
  const { ref } = useMove(setValue);

  const overlayStyles = { position: 'absolute', top: 0, left: 0, right: 0, bottom: 0 };

  return (
    <Group position="center">
      <div
        ref={ref}
        style={{
          width: 300,
          height: 150,
          backgroundColor: 'red',
          position: 'relative',
        }}
      >
        {/* Gradient overlays */}
        <div
          style={{ ...overlayStyles, backgroundImage: 'linear-gradient(90deg, #fff, transparent)' }}
        />
        <div
          style={{ ...overlayStyles, backgroundImage: 'linear-gradient(0deg, #000, transparent)' }}
        />
        <div
          style={{
            position: 'absolute',
            left: `calc(${value.x * 100}% - 8px)`,
            top: `calc(${value.y * 100}% - 8px)`,
            width: 16,
            height: 16,
            border: '2px solid #fff',
            borderRadius: 16,
          }}
        />
      </div>
    </Group>
  );
}
```

## Definition

```tsx
function useMove<T extends HTMLElement = HTMLDivElement>(
  onChange: (value: { x: number; y: number }) => void,
  handlers?: { onScrubStart?(): void; onScrubEnd?(): void }
): {
  ref: React.MutableRefObject<T>;
  active: boolean;
};
```
